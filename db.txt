create table criminaldata(
id INT AUTO_INCREMENT PRIMARY KEY,
    Name VARCHAR(255),
    `Father's Name` VARCHAR(255),
    `Mother's Name` VARCHAR(255),
    Gender VARCHAR(50),
    DOB DATE,
    `Blood Group` VARCHAR(10),
    `Identification Mark` TEXT,
    Nationality VARCHAR(100),
    Religion VARCHAR(100),
    `Crimes Done` TEXT
);
